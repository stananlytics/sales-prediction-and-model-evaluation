# Generate months and years automatically
months = list(range(1, 13)) * (len(sales)//12 + 1)
years = []
year_start = 2022
for y in range(year_start, year_start + (len(sales)//12 + 1)):
    years.extend([y]*12)

# Trim to length of sales
months = months[:len(sales)]
years = years[:len(sales)]

# Create dataframe
df = pd.DataFrame({"Year": years, "Month": months, "Sales": sales})

# ----------------------
# Step 2: Feature Engineering
# ----------------------

# Cyclical encoding of months
df["Month_sin"] = np.sin(2 * np.pi * df["Month"] / 12)
df["Month_cos"] = np.cos(2 * np.pi * df["Month"] / 12)

# ----------------------
# Step 3: Train/Test Split (simple)
# ----------------------
X = df[["Year", "Month_sin", "Month_cos"]]
y = df["Sales"]

# ----------------------
# Step 4: Linear Regression
# ----------------------
X_const = sm.add_constant(X)  # statsmodels needs constant
model_lr = sm.OLS(y, X_const).fit()

# Metrics
y_pred_lr = model_lr.predict(X_const)
mse_lr = mean_squared_error(y, y_pred_lr)
mae_lr = mean_absolute_error(y, y_pred_lr)
r2_lr = r2_score(y, y_pred_lr)

print("Linear Regression Results:")
print(f"AIC: {model_lr.aic:.2f}, BIC: {model_lr.bic:.2f}, R: {r2_lr:.4f}, MSE: {mse_lr:.2f}, MAE: {mae_lr:.2f}")

# ----------------------
# Step 5: Random Forest
# ----------------------
rf = RandomForestRegressor(n_estimators=500, random_state=42)
rf.fit(X, y)
y_pred_rf = rf.predict(X)

mse_rf = mean_squared_error(y, y_pred_rf)
mae_rf = mean_absolute_error(y, y_pred_rf)
r2_rf = r2_score(y, y_pred_rf)

def aic_bic_from_residuals(y_true, y_pred, k):
    n = len(y_true)
    resid = y_true - y_pred
    rss = float(np.sum(resid**2))
    sigma2 = rss / n
    # log-likelihood under N(0, sigma2)
    ll = -0.5 * n * (math.log(2 * math.pi * sigma2) + 1)
    AIC = 2 * k - 2 * ll
    BIC = math.log(n) * k - 2 * ll
    return AIC, BIC
